def serve_page(request, full_slug):
    is_authenticated = has_user_signed_up(request.user)
    slugs = full_slug.split('/')
    breadcrumbs = get_breadcrumbs(slugs)

    if not breadcrumbs:
        raise Http404

    current_page = breadcrumbs[-1]
    root_page = breadcrumbs[0]

    template = get_template(current_page)
    navigation = get_navigation(root_page,
        is_authenticated=is_authenticated)
    descendants = get_descendants(current_page)
    siblings = get_siblings(current_page)

    return render_to_response(template, {
        'navigation': navigation,
        'breadcrumbs': breadcrumbs,
        'page': current_page.as_leaf_class(),
        'siblings': siblings,
        'descendants': descendants,
        'title': current_page.title,
        'meta_keywords': current_page.meta_keywords,
        'meta_description': current_page.meta_description,
    }, context_instance=RequestContext(request))
